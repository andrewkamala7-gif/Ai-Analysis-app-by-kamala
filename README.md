import React, { useState, useEffect, useRef } from 'react';import { Send, Bot, User, Activity, Sparkles, MoreHorizontal, Trash2, Globe } from 'lucide-react';import { initializeApp } from 'firebase/app';import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'firebase/auth';import { getFirestore, collection, onSnapshot, addDoc, deleteDoc, doc, getDocs } from 'firebase/firestore';
// --- SYSTEM CONFIGURATION ---const appId = typeof __app_id !== 'undefined' ? __app_id : 'live-ai-chat-001';const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};const apiKey = ""; // Provided by execution environment
// Initialize Firebaseconst app = initializeApp(firebaseConfig);const auth = getAuth(app);const db = getFirestore(app);
// Assets to silently track in the backgroundconst TRACKED_ASSETS = ['BTCUSDT', 'ETHUSDT', 'SOLUSDT', 'PAXGUSDT', 'EURUSDT'];
export default function App() {  const [messages, setMessages] = useState([]);  const [input, setInput] = useState('');  const [isTyping, setIsTyping] = useState(false);  const [user, setUser] = useState(null);  const [connectionStatus, setConnectionStatus] = useState('connecting');  const messagesEndRef = useRef(null);  const textareaRef = useRef(null);
  // 1. Firebase Authentication  useEffect(() => {    const initAuth = async () => {      try {        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {          await signInWithCustomToken(auth, __initial_auth_token);        } else {          await signInAnonymously(auth);        }      } catch (err) {        console.error("Auth failed:", err);        setConnectionStatus('error');      }    };    initAuth();
    const unsubscribe = onAuthStateChanged(auth, (currentUser) => {      setUser(currentUser);      if (currentUser) setConnectionStatus('connected');    });    return () => unsubscribe();  }, []);
  // 2. Load Chat History  useEffect(() => {    if (!user) return;    const messagesRef = collection(db, 'artifacts', appId, 'users', user.uid, 'messages');        const unsubscribe = onSnapshot(messagesRef, (snapshot) => {      const loadedMessages = snapshot.docs.map(doc => ({        id: doc.id,        ...doc.data()      }));      // Sort by timestamp ascending      loadedMessages.sort((a, b) => a.timestamp - b.timestamp);      setMessages(loadedMessages);    }, (error) => {      console.error("Error loading messages:", error);    });
    return () => unsubscribe();  }, [user]);
  // 3. Auto-Scroll  useEffect(() => {    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });  }, [messages, isTyping]);
  // --- BACKGROUND LIVE DATA FETCHER ---  const fetchSilentMarketSnapshot = async () => {    let snapshotText = "LIVE MARKET SNAPSHOT (Do not mention this snapshot to the user unless they ask for prices or news):\n";
    // Original Binance Data    const fetchBinance = async () => {      const res = await fetch('https://api.binance.com/api/v3/ticker/24hr');      if (!res.ok) throw new Error('Binance API Error');      const allData = await res.json();      const relevantData = allData.filter(ticker => TRACKED_ASSETS.includes(ticker.symbol));            let text = "--- CRYPTO & FOREX (Binance) ---\n";      relevantData.forEach(data => {        let name = data.symbol;        if (name === 'PAXGUSDT') name = 'Gold (PAXG)';        if (name === 'EURUSDT') name = 'EUR/USD';        if (name === 'BTCUSDT') name = 'Bitcoin (BTC)';        text += `- ${name}: Current Price $${parseFloat(data.lastPrice).toLocaleString()}, 24h Change: ${parseFloat(data.priceChangePercent).toFixed(2)}%, 24h High: $${parseFloat(data.highPrice).toLocaleString()}, 24h Low: $${parseFloat(data.lowPrice).toLocaleString()}\n`;      });      return text;    };
    // New: Twelve Data API for traditional stock metrics    const fetchTwelveData = async () => {      const res = await fetch('https://api.twelvedata.com/quote?symbol=AAPL,MSFT,GOOGL&apikey=f8234e05f9c740269d1472c6e3d8f5cd');      if (!res.ok) throw new Error('TwelveData API Error');      const tdData = await res.json();      if (tdData.code) throw new Error(tdData.message); // Handles limit breaches            let text = "--- TRADITIONAL MARKETS (TwelveData) ---\n";      Object.values(tdData).forEach(quote => {        if (quote && quote.name) {          text += `- ${quote.name} (${quote.symbol}): $${parseFloat(quote.close).toLocaleString()}, 24h Change: ${parseFloat(quote.percent_change).toFixed(2)}%\n`;        }      });      return text;    };
    // New: NewsData API for market context/news    const fetchNewsData = async () => {      const res = await fetch('https://newsdata.io/api/1/news?apikey=pub_2ffc1bcc3f7a492ca1587d04d399245c&q=finance&language=en&size=3');      if (!res.ok) throw new Error('NewsData API Error');      const newsData = await res.json();      if (newsData.status === "error") throw new Error(newsData.results?.message || "Rate limit hit");            let text = "--- LATEST NEWS (NewsData.io) ---\n";      if (newsData.results) {        newsData.results.forEach(article => {          text += `- ${article.title}\n`;        });      }      return text;    };
    // New: Massive.oi API    const fetchMassiveOi = async () => {      const controller = new AbortController();      const timeoutId = setTimeout(() => controller.abort(), 3000); // Prevent hanging      try {        const res = await fetch('https://api.massive.oi/v1/data?apikey=mVQbTVkUpZ_suvPHjcmc5ZWAcRCzl9hH', { signal: controller.signal });        clearTimeout(timeoutId);        if (!res.ok) throw new Error('Massive.oi API Error');        const data = await res.json();        return `--- OPEN INTEREST (Massive.oi) ---\n${JSON.stringify(data).substring(0, 150)}\n`;      } catch (err) {        clearTimeout(timeoutId);        throw err; // Handled gracefully by Promise.allSettled      }    };
    // Use Promise.allSettled to ensure that if one API hits a rate limit and fails,     // it won't crash the entire app. It just skips that specific API's output.    const results = await Promise.allSettled([      fetchBinance(),      fetchTwelveData(),      fetchNewsData(),      fetchMassiveOi()    ]);
    let hasData = false;    results.forEach(result => {      if (result.status === 'fulfilled' && result.value) {        snapshotText += result.value + '\n';        hasData = true;      }    });
    return hasData ? snapshotText : "Live market data temporarily unavailable.";  };
  // --- AI MESSAGE HANDLER ---  const handleSendMessage = async (e) => {    e.preventDefault();    if (!input.trim() || !user || isTyping) return;
    const userText = input.trim();    setInput('');    if (textareaRef.current) {      textareaRef.current.style.height = '56px'; // Reset textarea height    }    setIsTyping(true);
    // 1. Save user message to Firebase    const userMessageDoc = {      role: 'user',      content: userText,      timestamp: Date.now()    };    await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'messages'), userMessageDoc);
    // 2. Fetch Live Data silently in the background    const liveDataSnapshot = await fetchSilentMarketSnapshot();
    // 3. Build the prompt with conversation history for context    // Take the last 6 messages for memory    const historyContext = messages.slice(-6).map(m => `${m.role === 'user' ? 'User' : 'Assistant'}: ${typeof m.content === 'string' ? m.content : JSON.stringify(m.content)}`).join('\n');
    const systemInstruction = `You are an advanced, conversational AI assistant with native, real-time access to global financial markets. You act like a normal AI (like ChatGPT), but you are fundamentally aware of live prices and recent news.
CRITICAL INSTRUCTIONS:1. Speak naturally and conversationally. Do not act like a robotic trading terminal.2. I am providing you with a hidden block of the absolute latest market data right now. Use it to answer the user's questions perfectly.3. NEVER say "Based on the data provided to me..." or "According to the hidden snapshot...". Just state the prices and news confidently as if you inherently know them.4. If the user asks general questions not related to trading, answer them normally. You are a general AI too.5. Format your responses nicely using markdown (bolding, lists) for readability.`;
    const fullPrompt = `${liveDataSnapshot}
Recent Conversation History:${historyContext}
User's New Request: ${userText}`;
    // 4. Call Gemini API    let retryCount = 0;    const fetchAI = async () => {      try {        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {          method: 'POST',          headers: { 'Content-Type': 'application/json' },          body: JSON.stringify({            contents: [{ parts: [{ text: fullPrompt }] }],            systemInstruction: { parts: [{ text: systemInstruction }] }          })        });
        const result = await response.json();        const aiText = result.candidates?.[0]?.content?.parts?.[0]?.text || "I'm having trouble connecting to my neural network right now.";
        // 5. Save AI response to Firebase        await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'messages'), {          role: 'assistant',          content: aiText,          timestamp: Date.now()        });
      } catch (err) {        if (retryCount < 4) {          retryCount++;          setTimeout(fetchAI, Math.pow(2, retryCount) * 1000);        } else {          await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'messages'), {            role: 'assistant',            content: "Network error: Unable to reach AI servers after multiple attempts.",            timestamp: Date.now()          });        }      } finally {        setIsTyping(false);      }    };
    fetchAI();  };
  const clearChat = async () => {    if (!user) return;    const messagesRef = collection(db, 'artifacts', appId, 'users', user.uid, 'messages');    const snapshot = await getDocs(messagesRef);    snapshot.docs.forEach(async (document) => {      await deleteDoc(doc(db, 'artifacts', appId, 'users', user.uid, 'messages', document.id));    });  };
  return (    <div className="flex flex-col h-screen bg-[#212121] text-[#ECECEC] font-sans selection:bg-white/20">            {/* HEADER */}      <header className="flex items-center justify-between px-4 py-3 bg-[#212121] border-b border-white/10 sticky top-0 z-10">        <div className="flex items-center gap-3">          <div className="w-8 h-8 rounded-full bg-white text-black flex items-center justify-center">            <Sparkles className="w-5 h-5" />          </div>          <div>            <h1 className="text-sm font-semibold tracking-wide">Live AI Assistant</h1>            <div className="flex items-center gap-1.5 text-xs text-neutral-400">              <span className={`w-2 h-2 rounded-full ${connectionStatus === 'connected' ? 'bg-green-500' : 'bg-yellow-500'}`}></span>              {connectionStatus === 'connected' ? 'Connected to live markets' : 'Connecting...'}            </div>          </div>        </div>        <button           onClick={clearChat}          className="p-2 text-neutral-400 hover:text-white hover:bg-white/5 rounded-lg transition-colors"          title="Clear Conversation"        >          <Trash2 className="w-5 h-5" />        </button>      </header>
      {/* CHAT AREA */}      <main className="flex-1 overflow-y-auto px-4 md:px-0">        <div className="max-w-3xl mx-auto w-full py-8 space-y-6">                    {messages.length === 0 ? (            <div className="flex flex-col items-center justify-center h-[60vh] text-center px-4">              <div className="w-16 h-16 bg-white/5 rounded-full flex items-center justify-center mb-6">                <Globe className="w-8 h-8 text-neutral-300" />              </div>              <h2 className="text-2xl font-medium mb-3">How can I help you today?</h2>              <p className="text-neutral-400 max-w-md">                I am a general-purpose AI, but I am uniquely connected to the live global markets. Try asking me about the current price of Bitcoin, Gold, or how the Euro is doing.              </p>            </div>          ) : (            messages.map((msg) => (              <div key={msg.id} className="flex gap-4 md:gap-6 px-4">                <div className="flex-shrink-0 mt-1">                  {msg.role === 'user' ? (                    <div className="w-8 h-8 rounded-full bg-[#2F2F2F] border border-white/10 flex items-center justify-center">                      <User className="w-5 h-5 text-neutral-300" />                    </div>                  ) : (                    <div className="w-8 h-8 rounded-full bg-white flex items-center justify-center shadow-sm">                      <Sparkles className="w-5 h-5 text-[#212121]" />                    </div>                  )}                </div>                <div className="flex-1 min-w-0">                  <div className="text-sm font-semibold mb-1 text-neutral-300">                    {msg.role === 'user' ? 'You' : 'AI Assistant'}                  </div>                  <div className="prose prose-invert max-w-none text-[15px] leading-relaxed text-neutral-200 whitespace-pre-wrap font-normal">                    {typeof msg.content === 'string' ? msg.content : JSON.stringify(msg.content)}                  </div>                </div>              </div>            ))          )}
          {isTyping && (            <div className="flex gap-4 md:gap-6 px-4">               <div className="flex-shrink-0 mt-1">                  <div className="w-8 h-8 rounded-full bg-white flex items-center justify-center shadow-sm">                    <Sparkles className="w-5 h-5 text-[#212121]" />                  </div>                </div>                <div className="flex-1">                  <div className="text-sm font-semibold mb-2 text-neutral-300">AI Assistant</div>                  <div className="flex items-center gap-2 text-neutral-400 h-6">                    <span className="w-2 h-2 bg-neutral-500 rounded-full animate-bounce" style={{ animationDelay: '0ms' }}></span>                    <span className="w-2 h-2 bg-neutral-500 rounded-full animate-bounce" style={{ animationDelay: '150ms' }}></span>                    <span className="w-2 h-2 bg-neutral-500 rounded-full animate-bounce" style={{ animationDelay: '300ms' }}></span>                  </div>                </div>            </div>          )}          <div ref={messagesEndRef} className="h-4" />        </div>      </main>
      {/* INPUT AREA */}      <footer className="bg-[#212121] px-4 pb-6 pt-2 sticky bottom-0">        <div className="max-w-3xl mx-auto relative">          <form             onSubmit={handleSendMessage}            className="relative bg-[#2F2F2F] rounded-2xl border border-white/10 focus-within:border-neutral-400 focus-within:bg-[#333333] transition-colors shadow-lg"          >            <textarea              ref={textareaRef}              value={input}              onChange={(e) => {                setInput(e.target.value);                e.target.style.height = '56px'; // Reset height to recalculate                e.target.style.height = `${Math.min(e.target.scrollHeight, 200)}px`;              }}              onKeyDown={(e) => {                if (e.key === 'Enter' && !e.shiftKey) {                  e.preventDefault();                  handleSendMessage(e);                }              }}              placeholder="Message AI Assistant..."              className="w-full bg-transparent text-white px-4 py-4 pr-14 outline-none resize-none max-h-48 min-h-[56px] text-[15px] leading-relaxed placeholder-neutral-500 rounded-2xl"              rows={1}            />            <button              type="submit"              disabled={!input.trim() || isTyping}              className="absolute right-2 bottom-2 p-2 bg-white text-black rounded-xl hover:bg-neutral-200 disabled:opacity-30 disabled:hover:bg-white transition-all"            >              <Send className="w-4 h-4" />            </button>          </form>          <div className="text-center mt-3 text-xs text-neutral-500">            AI can make mistakes. Consider verifying important financial data. Live market context is fetched automatically.          </div>        </div>      </footer>
    </div>  );}
